\documentclass[12pt, a4paper, oneside]{book}

%   a4paper - 纸张大小为A4
%   oneside - 单面打印（双面打印用twoside）

% 加载配置
\input{config/packages}      % 加载宏包配置文件
\input{config/settings}      % 加载格式设置文件
% \input{config/commands}      % 加载自定义命令文件

\begin{document}
% 文档内容开始

% ========== 前页部分 ==========
\frontmatter
% book类的特殊命令：表示前页部分（封面、摘要、目录等）
% 页码用罗马数字（i, ii, iii...）
% 中文封面
\include{frontmatter/cover-zh}  % 插入中文封面文件

\include{frontmatter/cover-en}  % 插入英文封面文件

\include{frontmatter/abstract-zh}  % 插入中文摘要
\include{frontmatter/abstract-en}  % 插入英文摘要
% 目录
\tableofcontents    % 生成目录
% \listoffigures      % 生成图目录（可选）
% \listoftables       % 生成表目录（可选）

% ========== 正文部分 ==========
% \frontmatter
\mainmatter
% book类的特殊命令：表示正文部分开始
% 页码用阿拉伯数字（1, 2, 3...），从1开始重新计数
% 各章节
\include{chapters/chapter01}  % 第1章
\include{chapters/chapter02}  % 第2章
\include{chapters/chapter03}  % 第3章

\backmatter

\end{document}
