% settings.tex - 格式设置文件
% 设置文档的全局格式，如页边距、标题样式、图表格式等
% 这里设置文档的全局格式参数

% 1. 页面设置
\geometry{
    a4paper,            % A4纸
    left=3cm,           % 左页边距
    right=2.5cm,        % 右页边距
    top=2.5cm,          % 上页边距
    bottom=2.5cm,       % 下页边距
    headheight=15pt     % 页眉高度
}

% 2. 行间距设置
\renewcommand{\baselinestretch}{1.5}  % 1.5倍行距
\setlength{\parskip}{0.5em}           % 段落间距

% 3. 章节标题格式
\titleformat{\chapter}[block]
    {\centering\Huge\bfseries}
    {第\ \thechapter\ 章}
    {1em}
    {}
\titlespacing*{\chapter}{0pt}{-20pt}{20pt}

% 4. 图表标题格式
\captionsetup[figure]{
    font=small,         % 字体大小
    labelfont=bf,       % 标签加粗
    labelsep=period,    % 标签后加句点
    justification=centering  % 居中
}
\captionsetup[table]{font=small, labelfont=bf}

% 5. 页眉页脚设置
\pagestyle{fancy}
\fancyhf{}  % 清除默认页眉页脚
\fancyhead[C]{\leftmark}  % 页眉居中显示章节名
\fancyfoot[C]{\thepage}   % 页脚居中显示页码
\renewcommand{\headrulewidth}{0.4pt}  % 页眉线宽度

% 6. 超链接设置
\hypersetup{
    colorlinks=true,        % 彩色链接
    linkcolor=blue,         % 内部链接颜色
    citecolor=green,        % 引用颜色
    urlcolor=magenta,       % URL颜色
    bookmarksnumbered=true  % 书签带编号
}

% 7. 代码列表设置
\lstset{
    basicstyle=\ttfamily\small,
    keywordstyle=\color{blue},
    commentstyle=\color{gray},
    numbers=left,
    numberstyle=\tiny,
    frame=single,
    breaklines=true
}
